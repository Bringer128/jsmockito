 25  *  <li><a href="#6">Making sure interactions never happened on a mock</a></li>

 28  *
 29  * <p>In the following examples object mocking is done with Array as this is
 30  * well understood, although you probably wouldn't mock this in normal test
 31  * development.</p>
 32  *
 33  * <h2><a name="1">1. Let's verify some behaviour!</a></h2>
 34  *
 35  * <p>For an object:</p>
 36  * <pre>
 37  * //mock creation
 38  * var mockedArray = mock(Array);
 39  *
 40  * //using mock object
 41  * mockedArray.push("one");
 42  * mockedArray.reverse();
 43  *
 44  * //verification
 45  * verify(mockedArray).push("one");
 46  * verify(mockedArray).reverse();
 47  * </pre>
 48  *
 49  * <p>For a function:</p>
 50  * <pre>
 51  * //mock creation
 52  * var mockedFunc = mockFunction();
 53  *
 54  * //using mock function
 55  * mockedFunc('hello world');
 56  * mockedFunc.call(this, 'foobar');
 57  * mockedFunc.apply(this, [ 'barfoo' ]);
 58  *
 59  * //verification
 60  * verify(mockedFunc)('hello world');
 61  * verify(mockedFunc)('foobar');
 62  * verify(mockedFunc)('barfoo');
 63  * </pre>
 64  *
 65  * <p>Once created a mock will remember all interactions.  Then you selectively
 66  * verify whatever interactions you are interested in.</p>
 67  *
 68  * <h2><a name="2">2. How about some stubbing?</a></h2>
 69  *
 70  * <p>For an object:</p>
 71  * <pre>
 72  * var mockedArray = mock(Array);
 73  *
 74  * //stubbing
 75  * when(mockedArray).slice(0).thenReturn('f');
 76  * when(mockedArray).slice(1).thenThrow('An exception');
 77  * when(mockedArray).slice(2).then(function() { return 1+2 });
 78  *
 79  * //the following returns "f"
 80  * assertThat(mockedArray.slice(0), equalTo('f'));
 81  *
 82  * //the following throws exception 'An exception'
 83  * var ex = undefined;
 84  * try {
 85  *   mockedArray.slice(1);
 86  * } catch (e) {
 87  *   ex = e;
 88  * }
 89  * assertThat(ex, equalTo('An exception');
 90  *
 91  * //the following invokes the stub method, which returns 3
 92  * assertThat(mockedArray.slice(2), equalTo(3));
 93  *
 94  * //the following returns undefined as slice(999) was not stubbed
 95  * assertThat(mockedArray.slice(999), typeOf('undefined'));
 96  *
 97  * //stubs can take multiple values to return in order (same for 'thenThrow' and 'then' as well)
 98  * when(mockedArray).pop().thenReturn('a', 'b', 'c');
 99  * assertThat(mockedArray.pop(), equalTo('a'));
100  * assertThat(mockedArray.pop(), equalTo('b'));
101  * assertThat(mockedArray.pop(), equalTo('c'));
102  * assertThat(mockedArray.pop(), equalTo('c'));
103  *
104  * //stubs can also be chained to return values in order
105  * when(mockedArray).unshift().thenReturn('a').thenReturn('b').then(function() { return 'c' });
106  * assertThat(mockedArray.unshift(), equalTo('a'));
107  * assertThat(mockedArray.unshift(), equalTo('b'));
108  * assertThat(mockedArray.unshift(), equalTo('c'));
109  * assertThat(mockedArray.unshift(), equalTo('c'));
110  *
111  * //stub matching can overlap, allowing for specific cases and defaults
112  * when(mockedArray).slice(3).thenReturn('abcde');
113  * when(mockedArray).slice(3, lessThan(0)).thenReturn('edcba');
114  * assertThat(mockedArray.slice(3, -1), equalTo('edcba'));
115  * assertThat(mockedArray.slice(3, 1), equalTo('abcde'));
116  * assertThat(mockedArray.slice(3), equalTo('abcde'));
117  *
118  * //can also verify a stubbed invocation, although this is usually redundant
119  * verify(mockedArray).slice(0);
120  * </pre>
121  *
122  * <p>For a function:</p>
123  * <pre>
124  * var mockedFunc = mockFunction();
125  *
126  * //stubbing
127  * when(mockedFunc)(0).thenReturn('f');
128  * when(mockedFunc)(1).thenThrow('An exception');
129  * when(mockedFunc)(2).then(function() { return 1+2 });
130  *
131  * //the following returns "f"
132  * assertThat(mockedFunc(0), equalTo('f'))
133  *
134  * //following throws exception 'An exception'
135  * mockedFunc(1);
136  * //the following throws exception 'An exception'
137  * var ex = undefined;
138  * try {
139  *   mockedFunc(1);
140  * } catch (e) {
141  *   ex = e;
142  * }
143  * assertThat(ex, equalTo('An exception');
144  *
145  * //the following invokes the stub method, which returns 3
146  * assertThat(mockedFunc(2), equalTo(3));
147  *
148  * //following returns undefined as mockedFunc(999) was not stubbed
149  * assertThat(mockedFunc(999), typeOf('undefined'));
150  *
151  * //stubs can take multiple values to return in order (same for 'thenThrow' and 'then' as well)
152  * when(mockedFunc)(3).thenReturn('a', 'b', 'c');
153  * assertThat(mockedFunc(3), equalTo('a'));
154  * assertThat(mockedFunc(3), equalTo('b'));
155  * assertThat(mockedFunc(3), equalTo('c'));
156  * assertThat(mockedFunc(3), equalTo('c'));
157  *
158  * //stubs can also be chained to return values in order
159  * when(mockedFunc)(4).thenReturn('a').thenReturn('b').then(function() { return 'c' });
160  * assertThat(mockedFunc(4), equalTo('a'));
161  * assertThat(mockedFunc(4), equalTo('b'));
162  * assertThat(mockedFunc(4), equalTo('c'));
163  * assertThat(mockedFunc(4), equalTo('c'));
164  *
165  * //stub matching can overlap, allowing for specific cases and defaults
166  * when(mockedFunc)(5).thenReturn('abcde')
167  * when(mockedFunc)(5, lessThan(0)).thenReturn('edcba')
168  * assertThat(mockedFunc(5, -1), equalTo('edcba'))
169  * assertThat(mockedFunc(5, 1), equalTo('abcde'))
170  * assertThat(mockedFunc(5), equalTo('abcde'))
171  *
172  * //can also verify a stubbed invocation, although this is usually redundant
173  * verify(mockedFunc)(0);
174  * </pre>
175  *
176  * <ul>
177  * <li>By default mocks return undefined from all invocations;</li>
178  * <li>Stubs can be overwritten;</li>
179  * <li>Once stubbed, the method will always return the stubbed value regardless
180  * of how many times it is called;</li>
181  * <li>Last stubbing is more important - when you stubbed the same method with
182  * the same (or overlapping) matchers many times.</li>
183  * </ul>
184  *
185  * <h2><a name="3">3. Matching Arguments</a></h2>
186  *
187  * <p>JsMockito verifies arguments using 
188  * <a href="http://jshamcrest.destaquenet.com/">JsHamcrest</a> matchers.
189  *
190  * <pre>
191  * var mockedArray = mock(Array);
192  * var mockedFunc = mockFunction();
193  *
194  * //stubbing using JsHamcrest
195  * when(mockedArray).slice(lessThan(10)).thenReturn('f');
196  * when(mockedFunc)(containsString('world')).thenReturn('foobar');
197  *
198  * //following returns "f"
199  * mockedArray.slice(5);
200  *
201  * //following returns "foobar"
202  * mockedFunc('hello world');
203  *
204  * //you can also use matchers in verification
205  * verify(mockedArray).slice(greaterThan(4));
206  * verify(mockedFunc)(equalTo('hello world'));
207  *
208  * //if not specified then the matcher is anything(), thus either of these
209  * //will match an invocation with a single argument
210  * verify(mockedFunc)();
211  * verify(mockedFunc)(anything());
212  * </pre>
213  *
214  * <ul>
215  * <li>If the argument provided during verification/stubbing is not a
216  * JsHamcrest matcher, then 'equalTo(arg)' is used instead;</li>
217  * <li>Where a function/method was invoked with an argument, but the stub or
218  * verification does not provide a matcher, then anything() is assumed;</li>
219  * <li>The reverse, however, is not true - the anything() matcher will
220  * <em>not</em> match an argument that was never provided.</li>
221  * </ul>
222  *
223  * <h2><a name="4">4. Verifying exact number of invocations / at least once /
224  * never</a></h2>
225  *
226  * <pre>
227  * var mockedArray = mock(Array);
228  * var mockedFunc = mockFunction();
229  *
230  * mockedArray.slice(5);
231  * mockedArray.slice(6);
232  * mockedFunc('a');
233  * mockedFunc('b');
234  *
235  * //verification of multiple matching invocations
236  * verify(mockedArray, times(2)).slice(anything());
237  * verify(mockedFunc, times(2))(anything());
238  *
239  * //the default is times(1), making these are equivalent
240  * verify(mockedArray, times(1)).slice(5);
241  * verify(mockedArray).slice(5);
242  * </pre>
243  *
244  * <h2><a name="5">5. Matching the context ('this')</a></h2>
245  * 
246  * Functions can be invoked with a specific context, using the 'call' or
247  * 'apply' methods. JsMockito mock functions (and mock object methods)
248  * will remember this context and verification/stubbing can match on it.
249  *
250  * <p>For a function:</p>
251  * <pre>
252  * var mockedFunc = mockFunction();
253  * var context1 = {};
254  * var context2 = {};
255  *
256  * when(mockedFunc).call(equalTo(context2), anything()).thenReturn('hello');
257  *
258  * mockedFunc.call(context1, 'foo');
259  * //the following returns 'hello'
260  * mockedFunc.apply(context2, [ 'bar' ]);
261  *
262  * verify(mockedFunc).apply(context1, [ 'foo' ]);
263  * verify(mockedFunc).call(context2, 'bar');
264  * </pre>
265  *
266  * <p>For object method invocations, the context is usually the object itself.
267  * But sometimes people do strange things, and you need to test it - so
268  * the same approach can be used for an object:</p>
269  * <pre>
270  * var mockedArray = mock(Array);
271  * var otherContext = {};
272  *
273  * when(mockedArray).slice.call(otherContext, 5).thenReturn('h');
274  *
275  * //the following returns 'h'
276  * mockedArray.slice.apply(otherContext, [ 5 ]);
277  *
278  * verify(mockedArray).slice.call(equalTo(otherContext), 5);
279  * </pre>
280  *
281  * <ul>
282  * <li>For mock functions, the default context matcher is anything();</li>
283  * <li>For mock object methods, the default context matcher is
284  * sameAs(mockObj).</li>
285  * </ul>
286  *
287  * <h2><a name="6">6. Making sure interactions never happened on a mock</a></h2>
288  * 
289  * <pre>
290  * var mockOne = mock(Array);
291  * var mockTwo = mock(Array);
292  * var mockThree = mockFunction();
293  * 
294  * //only mockOne is interacted with
295  * mockOne.push(5);
296  *
297  * //verify a method was never called
298  * verify(mockOne, never()).unshift('a');
299  * 
300  * //verify that other mocks were not interacted with
301  * verifyZeroInteractions(mockTwo, mockThree);
302  * </pre>
303  *
304  * <h2><a name="7">7. Finding redundant invocations</a></h2>
305  *
306  * <pre>
307  * var mockArray = mock(Array);
308  * 
309  * mockArray.push(5);
310  * mockArray.push(8);
311  *
312  * verify(mockArray).push(5);
313  *
314  * // following verification will fail
315  * verifyNoMoreInteractions(mockArray);
316  * </pre>
317  */

525 /**
526  * Create a mockable and stubbable anonymous function.
527  *
528  * <p>Once created, the function can be invoked and will return undefined for
529  * any interactions that do not match stub declarations.</p>
530  *
531  * <pre>
532  * var mockFunc = JsMockito.mockFunction();
533  * JsMockito.when(mockFunc).call(anything(), 1, 5).thenReturn(6);
534  * mockFunc(1, 5); // result is 6
535  * JsMockito.verify(mockFunc)(1, greaterThan(2));
536  * </pre>
537  *
538  * @param funcName {string} The name of the mock function to use in messages
539  *   (defaults to 'func')
540  * @param delegate {function} The function to delegate unstubbed calls to
541  *   (optional)
542  * @return {function} an anonymous function
543  */
544 JsMockito.mockFunction = function(funcName, delegate) {
545   if (typeof funcName == 'function') {
546     delegate = funcName;
547     funcName = undefined;
548   }
549   funcName = funcName || 'func';
550   delegate = delegate || function() { };
551 
552   var stubMatchers = []
553   var interactions = [];
554 
555   var mockFunc = function() {
556     var args = [this];
557     args.push.apply(args, arguments);
558     interactions.push({args: args});
559 
560     var stubMatcher = JsMockito.find(stubMatchers, function(stubMatcher) {
561       return JsMockito.matchArray(stubMatcher[0], args);
562     });
563     if (stubMatcher == undefined)
564       return delegate.apply(this, arguments);
565     var stubs = stubMatcher[1];
566     if (stubs.length == 0)
567       return undefined;
568     var stub = stubs[0];
569     if (stubs.length > 1)
570       stubs.shift();
571     return stub.apply(this, arguments);
572   };
573 
574   mockFunc.prototype = delegate.prototype;
575 
576   mockFunc._jsMockitoStubBuilder = function(contextMatcher) {
577     var contextMatcher = contextMatcher || JsHamcrest.Matchers.anything();
578     return matcherCaptureFunction(contextMatcher, function(matchers) {
579       var stubMatch = [matchers, []];
580       stubMatchers.unshift(stubMatch);
581       return {
582         then: function() {
583           stubMatch[1].push.apply(stubMatch[1], arguments);
584           return this;
585         },
586         thenReturn: function() {
587           return this.then.apply(this,JsMockito.map(arguments, function(value) {
588             return function() { return value };
589           }));
590         },
591         thenThrow: function(exception) {
592           return this.then.apply(this,JsMockito.map(arguments, function(value) {
593             return function() { throw value };
594           }));
595         }
596       };
597     });
598   };
599 
600   mockFunc._jsMockitoVerifier = function(verifier, contextMatcher) {
601     var contextMatcher = contextMatcher || JsHamcrest.Matchers.anything();
602     return matcherCaptureFunction(contextMatcher, function(matchers) {
603       return verifier(funcName, interactions, matchers, matchers[0] != contextMatcher);
604     });
605   };
606 
607   mockFunc._jsMockitoMockFunctions = function() {
608     return [ mockFunc ];
609   };
610 
611   return mockFunc;
612 
613   function matcherCaptureFunction(contextMatcher, handler) {
614     return JsMockito.contextCaptureFunction(contextMatcher,
615       function(context, args) {
616         var matchers = JsMockito.mapToMatchers([context].concat(args || []));
617         return handler(matchers);
618       }
619     );
620   };
621 };
622 JsMockito._export.push('mockFunction');

625 /**
626  * Create a mockable and stubbable objects.
627  *
628  * <p>A mock is created with the constructor for an object as an argument.
629  * Once created, the mock object will have all the same methods as the source
630  * object which, when invoked, will return undefined by default.</p>
631  *
632  * <p>Stub declarations may then be made for these methods to have them return
633  * useful values or perform actions when invoked.</p>
634  *
635  * <pre>
636  * MyObject = function() {
637  *   this.add = function(a, b) { return a + b }
638  * };
639  *
640  * var mockObj = JsMockito.mock(MyObject);
641  * mockObj.add(5, 4); // result is undefined
642  *
643  * JsMockito.when(mockFunc).add(1, 2).thenReturn(6);
644  * mockObj.add(1, 2); // result is 6
645  *
646  * JsMockito.verify(mockObj).add(1, greaterThan(2)); // ok
647  * JsMockito.verify(mockObj).add(1, equalTo(2)); // ok
648  * JsMockito.verify(mockObj).add(1, 4); // will throw an exception
649  * </pre>
650  *
651  * @param Obj {function} the constructor for the object to be mocked
652  * @return {object} a mock object
653  */
654 JsMockito.mock = function(Obj) {
655   var delegate = {};
656   if (typeof Obj != "function") {
657     delegate = Obj;
658     Obj = function() { };
659     Obj.prototype = delegate; 
660     Obj.prototype.constructor = Obj;
661   }
662   var MockObject = function() { };
663   MockObject.prototype = new Obj;
664   MockObject.prototype.constructor = MockObject;
665 
666   var mockObject = new MockObject();
667   var stubBuilders = {};
668   var verifiers = {};
669   
670   var contextMatcher = JsHamcrest.Matchers.sameAs(mockObject);
671 
672   var addMockMethod = function(name) {
673     var delegateMethod;
674     if (delegate[name] != undefined) {
675       delegateMethod = function() {
676         var context = (this == mockObject)? delegate : this;
677         return delegate[name].apply(context, arguments);
678       };
679     }
680     mockObject[name] = JsMockito.mockFunction('obj.' + name, delegateMethod);
681     stubBuilders[name] = mockObject[name]._jsMockitoStubBuilder;
682     verifiers[name] = mockObject[name]._jsMockitoVerifier;
683   };
684 
685   for (var methodName in mockObject) {
686     if (methodName != 'constructor')
687       addMockMethod(methodName);
688   }
689 
690   for (var typeName in JsMockito.nativeTypes) {
691     if (mockObject instanceof JsMockito.nativeTypes[typeName].type) {
692       JsMockito.each(JsMockito.nativeTypes[typeName].methods, function(method) {
693         addMockMethod(method);
694       });
695     }
696   }
697 
698   mockObject._jsMockitoStubBuilder = function() {
699     return JsMockito.mapInto(new MockObject(), stubBuilders, function(method) {
700       return method.call(this, contextMatcher);
701     });
702   };
703 
704   mockObject._jsMockitoVerifier = function(verifier) {
705     return JsMockito.mapInto(new MockObject(), verifiers, function(method) {
706       return method.call(this, verifier, contextMatcher);
707     });
708   };
709 
710   mockObject._jsMockitoMockFunctions = function() {
711     return JsMockito.objectValues(
712       JsMockito.mapObject(mockObject, function(func) {
713         return JsMockito.isMock(func)? func : null;
714       })
715     );
716   };
717 
718   return mockObject;

842 JsMockito.Verifier = function() { this.init.apply(this, arguments) };
843 JsMockito.Verifier.prototype = {
844   init: function() { },
845 
846   verify: function(mock) {
847     var self = this;
848     return mock._jsMockitoVerifier(function() {
849       self.verifyInteractions.apply(self, arguments);
850     });
851   },
852 
853   verifyInteractions: function(funcName, interactions, matchers, describeContext) {
854   },
855 
856   updateVerifiedInteractions: function(interactions) {
857     JsMockito.each(interactions, function(interaction) {
858       interaction.verified = true;
859     });
860   },
861 
862   buildDescription: function(message, funcName, matchers, describeContext) {
863     var description = new JsHamcrest.Description();
864     description.append(message + ': ' + funcName + '(');
865     JsMockito.each(matchers.slice(1), function(matcher, i) {
866       if (i > 0)
867         description.append(', ');
868       description.append('<');
869       matcher.describeTo(description);
870       description.append('>');
871     });
872     description.append(")");
873     if (describeContext) {
874       description.append(", 'this' being ");
875       matchers[0].describeTo(description);
876     }
877     return description;
878   }
879 };

881 JsMockito.verifier('Times', {
882   init: function(wanted) {
883     this.wanted = wanted;
884   },
885 
886   verifyInteractions: function(funcName, allInteractions, matchers, describeContext) {
887     var interactions = JsMockito.grep(allInteractions, function(interaction) {
888       return JsMockito.matchArray(matchers, interaction.args);
889     });
890     if (interactions.length == this.wanted) {
891       this.updateVerifiedInteractions(interactions);
892       return;
893     }
894 
895     var message;
896     if (interactions.length == 0) {
897       message = 'Wanted but not invoked';
898     } else if (this.wanted == 0) {
899       message = 'Never wanted but invoked';
900     } else if (this.wanted == 1) {
901       message = 'Wanted 1 invocation but got ' + interactions.length;
902     } else {
903       message = 'Wanted ' + this.wanted + ' invocations but got ' + interactions.length;
904     }
905 
906     var description = this.buildDescription(message, funcName, matchers, describeContext);
907     throw description.get();
908   }
909 });

921   verify: function(mock) {
922     var self = this;
923     JsMockito.each(mock._jsMockitoMockFunctions(), function(mockFunc) {
924       JsMockito.Verifier.prototype.verify.call(self, mockFunc)();
925     });

928   verifyInteractions: function(funcName, allInteractions, matchers, describeContext) {
929     var interactions = JsMockito.grep(allInteractions, function(interaction) {
930       return interaction.verified != true;
931     });
932     if (interactions.length == 0)
933       return;
934     
935     var description = this.buildDescription(
936       "No interactions wanted, but " + interactions.length + " remains",
937       funcName, matchers, describeContext);
938     throw description.get();
939   }
